
bash команды:
	- cd ... - перейти в директорию
	- ls - показать какие файлы есть в директории
	- ls -la - подробная информация о файлах
_____________________________________________________________

git команды:
	- git clone ... - клонировать репозиторий
	- git branch - показать ветки (локально)
	- git branch -a - показать все ветки (удаленные)
	- git branch -d "name branch" - удалить ветку
	- git branch -m "name branch" - переименовать ветку
	- git checkout "name branch" - перейти на ветку
	- git checkout "path file" - удалить изменения в этом файле
	- git checkout -b "name branch" - создать новую ветку
	- git checkout -b "name branch" "name of remote/name branch"
	- git add . - добавить в индекс (все файлы)
	- git commit -m - закоментировать изменения
	- git status - показать статус
	- git diff - показать побробные изменения
	- git log - показать изменения (комментарии и тп)
	- git fetch - показать удаленные изменения
	- git pull - скачать изменения с удаленного репозитория
	- git push - отправить локальные изменения в удаленный репозиторий
	- git merge "name branch" - смержить ветку
	- git stash - убрать все изменения (временно)
	- git stash pop - добавить временно убранные изменения
